# 监听文件打开
在`app.on('ready')`事件中添加以下内容：
```ts
function initFileOpen() {
    const handleOpen = async (eventName: string, path: string) => {
        appendOtdImportLog(`${eventName}    ${path}`);
        // 获取文件路径
        const filePath = decodeURIComponent(path);
        openMainWin('otd-open', filePath, `localStorage.setItem("OTD_OPEN_PATH", "${encodeURIComponent(filePath)}")`);
    };
    appendOtdImportLog(`initFileOpen`);

    const openPath = process && process.argv && process.argv.length ? process.argv.find((v) => v.endsWith('.otd')) : '';
    appendOtdImportLog(`${openPath}`);
    appendOtdImportLog(`${JSON.stringify(process.argv)}`);
    // 启动应用程序时有文件路径参数则执行打开操作
    if (openPath) {
        const timer = setInterval(() => {
            const wins = BrowserWindow.getAllWindows();
            if (wins.findIndex(win => win.webContents.getURL().includes('login.html')) !== -1) {
                handleOpen('启动应用',openPath)
                clearInterval(timer)
            }
        },100)
    }

    // 监听启动事件
    app.on('open-file', (event, filePath) => {
        event.preventDefault();
        filePath.endsWith('.otd') && handleOpen('open-file', filePath);
    });
    app.on('open-url', (event, url) => {
        event.preventDefault();
        url.endsWith('.otd') && handleOpen('open-url', url);
    });
    app.on('second-instance', (_, argv) => {
        appendOtdImportLog(JSON.stringify(argv));
        const path = argv.find((v) => v.endsWith('.otd'));
        path && handleOpen('second-instance', path);
    });
}
```
